static Square[][] createSquares(String diagram) {
        Square[][] grid = createSquares();
        int value = 0;
        for (int i = 0; i<9; i++) {
            for (int j = 0; j<9; j++) {
                if (diagram.charAt(value) != '.') {
                    int number = diagram.charAt(value) - 48;
                    grid[j][i].value = number;
                    value++;
                }
                value++;
            }
        }
        
        return grid;
    }
